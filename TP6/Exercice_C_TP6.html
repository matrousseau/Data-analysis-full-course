df_useconomics = pd.read_csv('useconomis.csv')

df_useconomics['M1'] = np.log(df_useconomics['M1'])
df_useconomics['gnp'] = np.log(df_useconomics['gnp'])
df_useconomics.columns=['log(M1)','log(gnp)','rd','rb']
logPNB = df_useconomics['log(gnp)']


def build_diff(df):
  
  diff = np.zeros(df.shape[0]-1)
  
  for i in range (1,len(diff)):
    
    diff[i-1] = df.iloc[i]-df.iloc[i-1]
    

  return diff

DiffPNB = build_diff(logPNB)
DiffPNB = pd.DataFrame(DiffPNB)
DiffPNB.columns = ['Diffrence']

DiffPNB = np.array(DiffPNB)

import warnings
from statsmodels.tsa.arima_model import ARIMA

warnings.filterwarnings('ignore')

X = DiffPNB

size = int(len(X) * 0.66)
train, test = X[0:size], X[size:len(X)]
history = [x for x in train]
predictions = list()
for t in range(len(test)):
  model = ARIMA(history, order=(0,1,2))
  model_fit = model.fit(disp=0)
  output = model_fit.forecast()
  yhat = output[0]
  predictions.append(yhat)
  obs = test[t]
  history.append(obs)
error = mean_squared_error(test, predictions)
  
print(model_fit.summary())
plt.plot(test)
plt.plot(predictions, color='red')
plt.show()
